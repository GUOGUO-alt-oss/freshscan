## 1. 设计原则

1. **编译即文档**：每个 public 方法必须有 KDoc、`@param`、`@return`、`@throws`
2. **状态机显式化**：ViewModel 状态使用 `sealed interface`，不允许隐式布尔组合
3. **错误不可忽略**：所有外部调用（相机、模型、数据库）必须有 try-catch 并映射到 UI 状态
4. **资源按量付费**：模型仅在使用页持有引用，退出页即释放
5. **低端机优先**：所有性能决策以 4GB RAM / 低端 SoC 为基准

```kotlin
// data/inference/EfficientDetEngine.kt — 🆕 新建文件
package com.example.freshscan.data.inference

import android.content.Context
import android.graphics.Bitmap
import android.graphics.RectF
import com.google.mediapipe.tasks.vision.core.RunningMode
import com.google.mediapipe.tasks.vision.objectdetector.ObjectDetector
import com.google.mediapipe.tasks.vision.objectdetector.ObjectDetectorResult
import javax.inject.Inject
import javax.inject.Singleton

/**
 * EfficientDet-Lite0 检测引擎封装。
 *
 * 使用 MediaPipe Tasks Vision ObjectDetector API。
 * 输入: Bitmap (自动 resize 到模型需求尺寸 320×320)
 * 输出: List<Detection> — 归一化坐标 [0,1] + COCO 类别索引 + 置信度
 *
 * 注意: MediaPipe Tasks Vision 内部捆绑了 TFLite runtime。
 * 应避免与独立 tensorflow-lite 依赖产生符号冲突。
 */
@Singleton
class EfficientDetEngine @Inject constructor(
    @ApplicationContext private val context: Context
) {
    private var objectDetector: ObjectDetector? = null

    // 检测配置
    companion object {
        const val SCORE_THRESHOLD = 0.4f      // 置信度阈值
        const val MAX_RESULTS = 10             // 最大检测框数
        const val MODEL_PATH = "efficientdet_lite0.tflite"
    }

    /**
     * 初始化检测器（延迟加载）。
     * 首次调用拍照时加载，2s 粒子动画覆盖加载时间。
     */
    fun ensureLoaded() {
        if (objectDetector != null) return
        objectDetector = ObjectDetector.createFromFileAndOptions(
            context,
            MODEL_PATH,
            ObjectDetector.ObjectDetectorOptions.builder()
                .setBaseOptions(
                    com.google.mediapipe.tasks.core.BaseOptions.builder()
                        .setModelAssetPath(MODEL_PATH)
                        .build()
                )
                .setRunningMode(RunningMode.IMAGE)
                .setScoreThreshold(SCORE_THRESHOLD)
                .setMaxResults(MAX_RESULTS)
                .build()
        )
    }

    /**
     * 执行检测。
     *
     * @param bitmap 输入图片（任意尺寸，MediaPipe 内部 resize）
     * @return ObjectDetectorResult 包含检测框列表
     */
    fun detect(bitmap: Bitmap): ObjectDetectorResult {
        ensureLoaded()
        val mpImage = com.google.mediapipe.framework.image.BitmapImageBuilder(bitmap).build()
        return objectDetector!!.detect(mpImage)
    }

    /**
     * 释放检测器资源。
     */
    fun close() {
        objectDetector?.close()
        objectDetector = null
    }
}
```

### 3.2 检测后处理（NMS + 坐标映射）

```kotlin
// data/inference/DetectionPostprocessor.kt — 🆕 新建文件
package com.example.freshscan.data.inference

import android.graphics.RectF
import com.google.mediapipe.tasks.vision.objectdetector.ObjectDetectorResult

/**
 * 检测框后处理：
 * 1. COCO 类别过滤（仅保留可能为果蔬的类别）
 * 2. NMS 去重（IoU 阈值 0.5）
 * 3. 坐标归一化 → 像素坐标映射
 */
object DetectionPostprocessor {

    /**
     * COCO 中与果蔬相关的类别索引。
     * 47=apple, 49=orange, 52=banana, 53=broccoli, 54=carrot, 55=hotdog,
     * 58=potted plant, 60=dining table (忽略), 70=bowl (忽略)
     *
     * 注意: EfficientDet 未在果蔬数据上微调，可能导致漏检。
     * 漏检时降级为滑动窗口方案（见 §3.6）。
     */
    val FOOD_CLASSES = setOf(47, 49, 52, 53, 54, 55)

    /**
     * 从 ObjectDetectorResult 中提取有效检测框。
     *
     * @param result MediaPipe 检测结果
     * @param imageWidth 原始图片宽度（用于坐标反归一化）
     * @param imageHeight 原始图片高度
     * @return 像素坐标的检测框列表，按置信度降序排列
     */
    fun extractValidBoxes(
        result: ObjectDetectorResult,
        imageWidth: Int,
        imageHeight: Int
    ): List<DetectedBox> {
        return result.detections()
            .filter { it.categories().any { cat -> cat.index() in FOOD_CLASSES } }
            .sortedByDescending { it.categories().maxOf { cat -> cat.score() } }
            .map { detection ->
                val bbox = detection.boundingBox()  // 归一化坐标 [0,1]
                DetectedBox(
                    rect = RectF(
                        bbox.left * imageWidth,
                        bbox.top * imageHeight,
                        bbox.right * imageWidth,
                        bbox.bottom * imageHeight
                    ),
                    confidence = detection.categories().maxOf { it.score() },
                    cocoClassId = detection.categories().first().index()
                )
            }
            .let { applyNms(it) }
    }

    /**
     * 简单 NMS（非极大值抑制）。
     *
     * 对已按置信度排序的框列表，移除与高置信框 IoU > 阈值 0.5 的低置信框。
     */
    fun applyNms(boxes: List<DetectedBox>, iouThreshold: Float = 0.5f): List<DetectedBox> {
        val kept = mutableListOf<DetectedBox>()
        val suppressed = BooleanArray(boxes.size) { false }

        for (i in boxes.indices) {
            if (suppressed[i]) continue
            kept.add(boxes[i])
            for (j in i + 1 until boxes.size) {
                if (!suppressed[j] && boxes[i].rect.iou(boxes[j].rect) > iouThreshold) {
                    suppressed[j] = true
                }
            }
        }
        return kept
    }
}

/**
 * 检测框的中间表示。
 */
data class DetectedBox(
    val rect: RectF,
    val confidence: Float,
    val cocoClassId: Int
)

/**
 * 两个矩形的 IoU（交并比）计算。
 */
fun RectF.iou(other: RectF): Float {
    val interLeft = maxOf(left, other.left)
    val interTop = maxOf(top, other.top)
    val interRight = minOf(right, other.right)
    val interBottom = minOf(bottom, other.bottom)
    if (interLeft >= interRight || interTop >= interBottom) return 0f

    val interArea = (interRight - interLeft) * (interBottom - interTop)
    val area1 = width() * height()
    val area2 = other.width() * other.height()
    val unionArea = area1 + area2 - interArea

    return if (unionArea <= 0f) 0f else interArea / unionArea
}
```

### 3.3 ModelMapperV2（260 类版本）

```kotlin
// data/mapper/ModelMapperV2.kt — 🆕 新建文件
package com.example.freshscan.data.mapper

import com.example.freshscan.data.inference.model.ModelConfigV2
import com.example.freshscan.domain.model.FreshnessLevel
import com.example.freshscan.util.Constants
import javax.inject.Inject
import javax.inject.Singleton
import kotlin.math.exp

/**
 * 将 Fruits-360 MobileNetV3-260 的 raw logits 映射为可读的标签信息。
 *
 * 与 v1 ModelMapper 的区别:
 * - 输出不再是 FruitCategory + FreshnessLevel，而是 label + displayName + isCookable
 * - 不包含新鲜度判断（新鲜度由独立模型负责）
 * - Top-3 改为 Top-5（260 类需要更多候选）
 */
@Singleton
class ModelMapperV2 @Inject constructor(
    private val config: ModelConfigV2
) {
    /**
     * @return Quadruple(label, displayName, confidence, isCookable)
     */
    fun mapToLabelInfo(rawOutput: FloatArray): LabelResult {
        val maxRawLogit = rawOutput.maxOrNull() ?: 0f
        if (maxRawLogit < Constants.MIN_LOGIT_FOR_CONFIDENCE) {
            return LabelResult.Unknown
        }

        val probs = softmax(rawOutput)
        val (topIndex, topConfidence) = probs
            .mapIndexed { i, p -> i to p }
            .maxByOrNull { it.second } ?: (0 to 0f)

        if (topConfidence < Constants.CONFIDENCE_THRESHOLD) {
            return LabelResult.Unknown
        }

        val labelInfo = config.labels.getOrNull(topIndex)
            ?: return LabelResult.Unknown

        return LabelResult.Known(
            label = labelInfo.label,
            displayName = labelInfo.displayName,
            confidence = topConfidence,
            isCookable = labelInfo.isCookable
        )
    }

    /**
     * Top-5 预测（用于调试和检查）。
     */
    fun mapToTop5(rawOutput: FloatArray): List<LabelResult.Known> {
        val probs = softmax(rawOutput)
        return probs
            .mapIndexed { i, p -> i to p }
            .sortedByDescending { it.second }
            .take(5)
            .mapNotNull { (i, p) ->
                config.labels.getOrNull(i)?.let {
                    LabelResult.Known(it.label, it.displayName, p, it.isCookable)
                }
            }
    }

    private fun softmax(logits: FloatArray): FloatArray {
        val maxLogit = logits.maxOrNull() ?: 0f
        val exps = FloatArray(logits.size) { exp((logits[it] - maxLogit).toDouble()).toFloat() }
        val sum = exps.sum()
        return if (sum == 0f || sum.isNaN()) FloatArray(logits.size) { 1f / logits.size }
        else FloatArray(logits.size) { exps[it] / sum }
    }
}

sealed interface LabelResult {
    data object Unknown : LabelResult
    data class Known(
        val label: String,
        val displayName: String,
        val confidence: Float,
        val isCookable: Boolean
    ) : LabelResult
}
```

### 3.5 三阶段推理管线（AnalysisViewModel 中的核心逻辑）

```kotlin
// 以下代码嵌入 AnalysisViewModel.infer(bitmap) 方法中
// （完整 ViewModel 见 §4.1）

private suspend fun runInference(bitmap: Bitmap): List<DetectedItem> = withContext(Dispatchers.Default) {

    // ═══ 阶段 1: 多目标检测 ═══
    val detectResult = efficientDet.detect(bitmap)
    val boxes = DetectionPostprocessor.extractValidBoxes(
        detectResult, bitmap.width, bitmap.height
    )

    // ═══ 降级检测: 如果 EfficientDet 无结果，启用滑动窗口 ═══
    val detectionBoxes = if (boxes.isEmpty()) {
        slidingWindowDetect(bitmap)  // §3.6
    } else {
        boxes
    }

    if (detectionBoxes.isEmpty()) return@withContext emptyList()

    // ═══ 阶段 2: 对每个框并行做品类识别 + 新鲜度分类 ═══
    detectionBoxes.map { box ->
        val crop = cropBitmap(bitmap, box.rect) ?: return@map null

        // 并行执行品类识别和新鲜度检测
        val labelDeferred = async { classifyCrop260(crop) }
        val freshnessDeferred = async { classifyCrop18(crop) }

        val labelResult = labelDeferred.await()
        val freshness = freshnessDeferred.await()

        labelResult?.let {
            DetectedItem(
                id = UUID.randomUUID().toString(),
                label = it.label,
                displayName = it.displayName,
                freshnessLevel = freshness,
                confidence = it.confidence,
                bbox = RectF(
                    box.rect.left / bitmap.width,
                    box.rect.top / bitmap.height,
                    box.rect.right / bitmap.width,
                    box.rect.bottom / bitmap.height
                ),
                isCookable = it.isCookable
            )
        }
    }.filterNotNull()
}

// 辅助方法
private suspend fun classifyCrop260(bitmap: Bitmap): LabelResult.Known? {
    val input = imagePreprocessor.bitmapToTensorBuffer(bitmap, 224, 224)
    val logits = classifier260.classify(input.buffer)
    return (modelMapper260.mapToLabelInfo(logits) as? LabelResult.Known)
}

private suspend fun classifyCrop18(bitmap: Bitmap): FreshnessLevel {
    val input = imagePreprocessor.bitmapToTensorBuffer(bitmap, 224, 224)
    val logits = classifierFreshness.classify(input.buffer)
    return modelMapperFreshness.mapToResult(logits, 0L).freshnessLevel
}

private fun cropBitmap(bitmap: Bitmap, rect: RectF): Bitmap? {
    val left = rect.left.toInt().coerceIn(0, bitmap.width)
    val top = rect.top.toInt().coerceIn(0, bitmap.height)
    val right = rect.right.toInt().coerceIn(0, bitmap.width)
    val bottom = rect.bottom.toInt().coerceIn(0, bitmap.height)
    if (right - left < 20 || bottom - top < 20) return null
    return Bitmap.createBitmap(bitmap, left, top, right - left, bottom - top)
}
```

### 3.6 滑动窗口降级检测

当 EfficientDet 检测结果为空时（如检测辣椒、秋葵等 COCO 不含的类别），降级为滑动窗口方案。

```kotlin
// 嵌入 AnalysisViewModel
private fun slidingWindowDetect(bitmap: Bitmap): List<DetectedBox> {
    // 策略: 固定窗口 + 粗粒度步长
    // 最大窗口数 = ceil(W/stride) × ceil(H/stride) ≈ 64 (对 1280px 图片)
    // 64 × 50ms ≈ 3.2s → 超出动画时间
    // 优化: 先降采样到 640px → 16 窗口 → 0.8s

    val downsampled = Bitmap.createScaledBitmap(bitmap, 640, 640, true)
    val windowSize = 224
    val stride = 224  // 无重叠

    val results = mutableListOf<Pair<RectF, LabelResult.Known>>()

    var y = 0
    while (y + windowSize <= downsampled.height) {
        var x = 0
        while (x + windowSize <= downsampled.width) {
            val crop = Bitmap.createBitmap(downsampled, x, y, windowSize, windowSize)
            val input = imagePreprocessor.bitmapToTensorBuffer(crop, 224, 224)
            val logits = classifier260.classify(input.buffer)
            val result = modelMapper260.mapToLabelInfo(logits)
            if (result is LabelResult.Known && result.confidence > 0.6f) {
                // 映射回原始图片坐标
                val scaleX = bitmap.width.toFloat() / 640f
                val scaleY = bitmap.height.toFloat() / 640f
                results.add(
                    RectF(x * scaleX, y * scaleY, (x + windowSize) * scaleX, (y + windowSize) * scaleY)
                        to result
                )
            }
            crop.recycle()
            x += stride
        }
        y += stride
    }
    downsampled.recycle()

    // 去重: 对分类结果做 NMS
    return DetectionPostprocessor.applyNms(
        results.map { (rect, label) ->
            DetectedBox(rect = rect, confidence = label.confidence, cocoClassId = -1)
        }
    )
}
```

### 3.7 推理时序（完整分析页生命周期）

```
[0ms] 拍照返回, AnalysisScreen 显示 → ActivityResultCallback
[0ms] viewModel.infer(bitmap) 启动 → 协程 Dispatchers.Default
[0ms] 粒子动画开始 → Canvas 持续渲染 2 秒
      ┌── 并行 ──┐
[50ms]   efficientDet.ensureLoaded() → 首次加载 ~200ms
[250ms]  efficientDet.detect(bitmap) → ~80ms
[330ms]  NMS 后处理 → ~5ms
[335ms]  for each box (假设 3 框):
           async { classify260(crop) } → ~50ms/框 (并行 ~50ms)
           async { classify18(crop) }  → ~50ms/框 (并行 ~50ms)
[450ms]  全部分类完成 → List<DetectedItem>
[450ms]  _uiState.update { isAnalyzing = false }
      ────────────
[2000ms] 粒子动画结束 → scaleTo(0) + fadeOut(300ms)
[2100ms] 如果有检测结果 → BottomSheet 展开到 HALF
         如果无检测结果 → BottomSheet 展开到 COLLAPSED + 提示重拍
```

```kotlin
// ui/screen/analysis/AnalysisViewModel.kt — 🆕 新建文件
@HiltViewModel
class AnalysisViewModel @Inject constructor(
    @ApplicationContext private val context: Context,
    private val savedStateHandle: SavedStateHandle,
    private val efficientDet: EfficientDetEngine,
    @ModelV2 private val classifier260: TFLiteClassifier,
    @FreshnessModel private val classifierFreshness: TFLiteClassifier,
    private val modelMapper260: ModelMapperV2,
    private val modelMapperFreshness: ModelMapper,       // v1 ModelMapper (新鲜度专用, 18类)
    private val imagePreprocessor: ImagePreprocessor,
    private val recipeEngine: RecipeEngine,

    val uiState: StateFlow<AnalysisUiState> = _uiState.asStateFlow()

    // SideEffect 通道（一次性事件，如导航）
    private val _sideEffects = Channel<AnalysisSideEffect>(Channel.BUFFERED)
    val sideEffects: Flow<AnalysisSideEffect> = _sideEffects.receiveAsFlow()

    init {
        // Process Death 恢复
        val savedUriString = savedStateHandle.get<String>("photoUri")
        if (savedUriString != null) {
            val uri = Uri.parse(savedUriString)
            _uiState.update { it.copy(photoUri = uri) }
            startAnalysis(uri)
        }
    }

    /**
     * 开始分析指定 URI 的图片。
     * 由 ActivityResultCallback 或 Process Death 恢复时调用。
     */
    fun startAnalysis(photoUri: Uri) {
        savedStateHandle["photoUri"] = photoUri.toString()
        _uiState.update { it.copy(
            photoUri = photoUri,
            screenState = if (isModelReady()) AnalysisScreenState.Animating
                          else AnalysisScreenState.Loading
        ) }

        viewModelScope.launch {
            try {
                // 延迟加载模型
                ensureModelsLoaded()
                _uiState.update { it.copy(screenState = AnalysisScreenState.Animating) }

                val bitmap = loadBitmap(photoUri)
                    ?: throw IllegalStateException("无法加载照片")

                val items = runInference(bitmap)  // §3.5

                val screenState = if (items.isEmpty()) {
                    AnalysisScreenState.Empty
                } else {
                    AnalysisScreenState.Results(items.size)
                }

                _uiState.update { it.copy(
                    items = items,
                    screenState = screenState,
                    isAnalyzing = false,
                    errorMessage = null
                ) }

                // 自动保存到历史（仅当有结果时）
                if (items.isNotEmpty()) {
                    saveToHistory(items)
                }

            } catch (e: Exception) {
                Logger.e("AnalysisViewModel", "Analysis failed", e)
                _uiState.update { it.copy(
                    screenState = AnalysisScreenState.Error(
                        e.message ?: "分析失败，请重试"
                    ),
                    isAnalyzing = false
                )}
            }
        }
    }

    /**
     * 根据检测结果推荐菜谱。
     * 用户在 BottomSheet 中点击 [找菜谱] 时调用。
     */
    fun findRecipes(category: RecipeCategory? = null) {
        viewModelScope.launch {
            val items = _uiState.value.items
            val profile = loadTasteProfile()
            val result = recipeEngine.recommend(items, profile)

            var recipes = result.recipes
            if (category != null) {
                recipes = recipes.filter { it.category == category }
            }

            _uiState.update { it.copy(
                recipes = recipes,
                recipeNote = result.note,
                selectedCategory = category,
                sheetState = if (recipes.isNotEmpty()) SheetState.FULL
                             else SheetState.COLLAPSED
            )}
        }
    }

    /**
     * 重拍：关闭当前分析，通知 UI 跳转拍照。
     */
    fun retake() {
        savedStateHandle.remove<Unit>("photoUri")
        _uiState.update { AnalysisUiState() }
        viewModelScope.launch {
            _sideEffects.send(AnalysisSideEffect.Retake)
        }
    }

    /**
     * 重试分析（错误恢复路径）。
     */
    fun retry() {
        val uri = _uiState.value.photoUri ?: return
        _uiState.update { it.copy(screenState = AnalysisScreenState.Loading, errorMessage = null) }
        startAnalysis(uri)
    }

    /**
     * 更新 BottomSheet 状态。
     */
    fun setSheetState(state: SheetState) {
        _uiState.update { it.copy(sheetState = state) }
    }

    override fun onCleared() {
        super.onCleared()
        savedStateHandle.remove<Unit>("photoUri")
        // 延迟 30 秒后关闭所有 Interpreter（避免频繁进出重加载）
        viewModelScope.launch {
            delay(30_000)
            closeAllModels()
        }
    }

    // ─── Private ───

    private fun isModelReady(): Boolean =
        classifier260.isLoaded() && classifierFreshness.isLoaded()

    private suspend fun ensureModelsLoaded() = withContext(Dispatchers.Default) {
        efficientDet.ensureLoaded()
        classifier260.ensureLoaded()
        classifierFreshness.ensureLoaded()
    }

    private fun closeAllModels() {
        efficientDet.close()
        classifier260.close()
        classifierFreshness.close()
    }

    private suspend fun loadBitmap(uri: Uri): Bitmap? = withContext(Dispatchers.IO) {
        val options = BitmapFactory.Options().apply { inSampleSize = 2 }
        val inputStream = context.contentResolver.openInputStream(uri)
        val bitmap = BitmapFactory.decodeStream(inputStream, null, options)
        inputStream?.close()
        // 限制最大尺寸为 1920px 防止 OOM
        bitmap?.let {
            if (maxOf(it.width, it.height) > 1920) {
                val scale = 1920f / maxOf(it.width, it.height)
                Bitmap.createScaledBitmap(
                    it,
                    (it.width * scale).toInt(),
                    (it.height * scale).toInt(),
                    true
                ).also { scaled -> if (scaled != it) it.recycle() }
            } else it
        }
    }

    // runInference() 见 §3.5
    // slidingWindowDetect() 见 §3.6
}
```

```kotlin
// data/recipe/RecipeEngine.kt — 匹配算法实现

fun recommend(items: List<DetectedItem>, profile: TasteProfile? = null): RecipeResult {
    // Step 1: 过滤可烹饪食材
    val cookable = items.filter { it.isCookable }
    if (cookable.isEmpty()) {
        return RecipeResult(emptyList(), "检测到的全是水果，建议直接食用 🍎")
    }

    // Step 2: 标签归一化
    val ingredientNames = cookable
        .flatMap { labelNormalizer.normalize(it.label) }
        .distinct()

    // Step 3: 按食材匹配数 + 口味档案加权排序
    val scored = presetRecipes.map { recipe ->
        val matchCount = recipe.matchIngredients.count { it in ingredientNames }
        val hasAllMatch = recipe.matchIngredients.all { it in ingredientNames }

        var score = matchCount.toFloat()

        // 完整匹配加权
        if (hasAllMatch) score += 3f

        // 口味档案加权
        if (profile != null) {
            val prefCategories = profile.preferredCategories
            if (recipe.category in prefCategories) score += 1.5f
        }

        recipe to score
    }.filter { (_, score) -> score > 0 }

    // Step 4: 排序（分数降序 → 烹饪时间升序）
    val sorted = scored
        .sortedWith(compareByDescending<Pair<Recipe, Float>> { it.second }
            .thenBy { it.first.cookingTimeMin })
        .map { it.first }

    return RecipeResult(sorted, null)
}
```

### 7.1 粒子动画数学

```kotlin
// ui/components/ParticleScan.kt — 🆕 新建文件
/**
 * 扫描粒子动画 Canvas。
 *
 * 物理模型：
 * - N = 80 个粒子（UI spec 原定 200，为保 60fps 降为 80；实际可配置）
 * - 暗色模式下 N = 60（降低 25% 减少 GPU 负载，UI spec §6 原定 150）
 * - 初始位置: 随机分布在屏幕 4 条边上
 * - 目标: 向照片中心移动
 * - 运动: 匀加速直线（a = 400px/s²）
 * - 生命周期: 生成后 2 秒内 alpha 从 1→0
 * - 颜色: 白色 + 金色系 (UI spec §2: #FFFFFF + #FFD54F)
 */

data class Particle(
    var x: Float,           // 当前 X
    var y: Float,           // 当前 Y
    val targetX: Float,     // 目标 X（照片中心）
    val targetY: Float,     // 目标 Y
    var vx: Float = 0f,     // X 方向速度
    var vy: Float = 0f,     // Y 方向速度
    var life: Float = 1f,   // 剩余生命 [0,1]
    val size: Float = 3f + Random.nextFloat() * 5f,   // 粒子大小 3-8dp
    val color: Color = ParticleColors[Random.nextInt(ParticleColors.size)]
)

/**
 * 粒子生命周期:
 *
 * [0ms]  生成 80 粒子在屏幕边缘
 * [0ms]  粒子开始向中心移动
 * [50ms] 每 50ms 生成新一批 10 粒子（补充衰减）
 * [100ms] 继续补充...
 * [1500ms] 停止生成新粒子
 * [2000ms] 所有粒子 alpha → 0，动画结束
 *
 * 每帧更新 (16.67ms @ 60fps):
 *   1. 更新每个粒子的速度和位置（加速向中心）
 *   2. 减少生命值（life -= dt / 2000ms）
 *   3. 绘制所有 life > 0 的粒子
 */

@Composable
fun ParticleScan(
    isActive: Boolean,
    photoBitmap: Bitmap?,
    modifier: Modifier = Modifier
) {
    val particles = remember { mutableStateListOf<Particle>() }
    val centerX = remember { mutableFloatStateOf(0f) }
    val centerY = remember { mutableFloatStateOf(0f) }
    var lastSpawnTime by remember { mutableLongStateOf(0L) }

    // 动画循环
    LaunchedEffect(isActive) {
        if (!isActive) return@LaunchedEffect

        while (isActive) {
            val now = withFrameMillis { it }

            // 生成新粒子（前 1.5 秒，每 50ms 一批）
            if (now - lastSpawnTime > 50 && now < 1500) {
                spawnParticles(particles, centerX.floatValue, centerY.floatValue, 10)
                lastSpawnTime = now
            }

            // 更新粒子
            val dt = 0.01667f  // 60fps
            val iterator = particles.iterator()
            while (iterator.hasNext()) {
                val p = iterator.next()
                // 计算加速度方向（指向中心）
                val dx = p.targetX - p.x
                val dy = p.targetY - p.y
                val dist = sqrt(dx * dx + dy * dy)
                if (dist > 1f) {
                    val ax = (dx / dist) * ACCELERATION
                    val ay = (dy / dist) * ACCELERATION
                    p.vx += ax * dt
                    p.vy += ay * dt
                }
                p.x += p.vx * dt
                p.y += p.vy * dt
                p.life -= dt / 2f  // 2 秒生命周期

                if (p.life <= 0f) iterator.remove()
            }

            // 初始生成（第一批）
            if (particles.isEmpty() && now < 100) {
                spawnParticles(particles, centerX.floatValue, centerY.floatValue, 80)
            }
        }
    }

    // 绘制
    Canvas(modifier = modifier.fillMaxSize()) {
        centerX.floatValue = size.width / 2f
        centerY.floatValue = size.height / 2f

        particles.forEach { p ->
            drawCircle(
                color = p.color.copy(alpha = p.life),
                radius = p.size,
                center = Offset(p.x, p.y)
            )
        }
    }

    // 动画结束时淡出
    AnimatedVisibility(
        visible = isActive,
        enter = fadeIn(300),
        exit = fadeOut(300)
    ) {
        // 保留 Canvas
    }
}

private fun spawnParticles(
    list: MutableList<Particle>,
    cx: Float, cy: Float, count: Int
) {
    repeat(count) {
        val (x, y) = randomEdgePosition(cx, cy)
        list.add(Particle(
            x = x, y = y,
            targetX = cx, targetY = cy,
            vx = (Random.nextFloat() - 0.5f) * 100f,
            vy = (Random.nextFloat() - 0.5f) * 100f
        ))
    }
}

/** 在屏幕 4 条边上随机生成位置。 */
private fun randomEdgePosition(cx: Float, cy: Float): Pair<Float, Float> {
    val edge = Random.nextInt(4)
    return when (edge) {
        0 -> Random.nextFloat() * cx * 2 to 0f                    // 顶部
        1 -> Random.nextFloat() * cx * 2 to cy * 2                // 底部
        2 -> 0f to Random.nextFloat() * cy * 2                    // 左侧
        else -> cx * 2 to Random.nextFloat() * cy * 2             // 右侧
    }
}

private const val ACCELERATION = 400f  // px/s²
private val ParticleColors = listOf(
    Color(0xFFFFFFFF).copy(alpha = 0.8f),  // 白色 (UI spec: #FFFFFF alpha 0.8)
    Color(0xFFFFF8E1).copy(alpha = 0.7f),  // 暖白
    Color(0xFFFFD54F).copy(alpha = 0.4f),  // 金色 (UI spec: #FFD54F alpha 0.4)
    Color(0xFFFFECB3).copy(alpha = 0.5f),  // 浅金色
    Color(0xFFFFFFFF).copy(alpha = 0.6f)   // 白色半透明
)
```

```kotlin
// ui/screen/analysis/AnalysisScreen.kt 中的 BottomSheet 配置
/**
 * ModalBottomSheet 三态 + 嵌套滚动冲突解决。
 *
 * Material3 ModalBottomSheet 原生仅支持两个可见状态:
 * - PartiallyExpanded (≈220dp) → 映射为 COLLAPSED
 * - Expanded (≈屏幕 60%) → 映射为 HALF
 *
 * FULL 态通过将 sheet 内容高度设为 fillMaxHeight() + 用户上滑触发展开实现。
 * 实际实现时可通过 BottomSheetScaffold（支持 3 态）替代 ModalBottomSheet 以获得
 * 更精确的 HALF/FULL 区分。
 *
 * 状态定义:
 * - COLLAPSED: peekHeight = 220.dp (显示结果摘要)
 * - HALF: 占屏幕 ~60% (ModalBottomSheet Expanded 态，显示结果 + 筛选)
 * - FULL: 内容 fillMaxHeight，实际高度取决于 LazyColumn 内容量
 *
 * 嵌套滚动冲突解决:
 * 问题: LazyColumn 在 BottomSheet 内部时，
 *       向下拖拽列表到顶后继续下拉会触发 BottomSheet 关闭。
 * 
 * 方案: NestedScrollConnection 拦截。
 *   - LazyColumn 未到顶 → 手势传递给 LazyColumn
 *   - LazyColumn 到顶 + 向下拖 → 手势传递给 BottomSheet
 *   - LazyColumn 到底 + 向上拖 → 手势传递给 BottomSheet
 */

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun AnalysisBottomSheet(
    sheetState: SheetState,
    items: List<DetectedItem>,
    recipes: List<Recipe>,
    onRecipeClick: (String) -> Unit,
    onRetake: () -> Unit,
    onStateChange: (SheetState) -> Unit,
    modifier: Modifier = Modifier
) {
    val modalSheetState = rememberModalBottomSheetState(
        skipPartiallyExpanded = false,
        confirmValueChange = { value ->
            // ModalBottomSheet SheetValue → 自定义 SheetState 映射
            // 注意: M3 ModalBottomSheet 仅有 PartiallyExpanded 和 Expanded 两个可见态
            // Hidden → 用户下滑关闭 → 回首页
            onStateChange(when (value) {
                SheetValue.Hidden -> SheetState.COLLAPSED        // 即将关闭
                SheetValue.PartiallyExpanded -> SheetState.COLLAPSED // ~220dp
                SheetValue.Expanded -> SheetState.HALF            // ~60% 屏幕
                // FULL 态通过内部 LazyColumn 滚动触发展开，不由 SheetValue 直接驱动
            })
            true
        }
    )

    ModalBottomSheet(
        onDismissRequest = { onRetake() },
        sheetState = modalSheetState,
        modifier = modifier.nestedScroll(rememberNestedScrollInteropConnection())
    ) {
        Column(modifier = Modifier.fillMaxWidth()) {
            // 结果头部
            ResultHeader(items = items, onFindRecipes = { /* 触发找菜谱 */ })

            // 菜谱列表（LazyColumn，滚动到顶部时允许 BottomSheet 拖拽）
            if (recipes.isNotEmpty()) {
                RecipeFilterChips()
                LazyColumn {
                    items(recipes) { recipe ->
                        RecipeCard(recipe, onClick = { onRecipeClick(recipe.id) })
                    }
                }
            }
        }
    }
}

/**
 * 嵌套滚动拦截连接。
 * 防止 LazyColumn 滚动到顶/底时触发 BottomSheet 的拖拽关闭。
 *
 * 这是 Compose Material3 ModalBottomSheet 的已知问题，
 * 需要显式处理 nestedScroll 的优先级。
 */
@Composable
fun rememberNestedScrollInteropConnection(): NestedScrollConnection {
    return remember {
        object : NestedScrollConnection {
            override fun onPreScroll(available: Offset, source: NestedScrollSource): Offset {
                // 如果 BottomSheet 未完全展开，且用户向上拖 → 让 BottomSheet 处理
                return if (source == NestedScrollSource.Drag) {
                    Offset.Zero  // 先让内容处理
                } else {
                    Offset.Zero
                }
            }
        }
    }
}
```

### 8.4 OOM 预防

```kotlin
// Bitmap 加载的安全策略

/**
 * 安全加载图片，防止 OOM。
 *
 * 策略:
 * 1. inSampleSize = 2 (默认降采样)
 * 2. 最大尺寸限制 1920px
 * 3. 加载完立即关闭 InputStream
 * 4. 推理完成后立即 recycle 非必需的 Bitmap
 */
fun loadBitmapSafely(context: Context, uri: Uri): Bitmap? {
    val options = BitmapFactory.Options().apply {
        inJustDecodeBounds = true
    }
    // 第一遍: 只读尺寸
    context.contentResolver.openInputStream(uri)?.use { stream ->
        BitmapFactory.decodeStream(stream, null, options)
    }

    // 计算合适的采样率
    options.inSampleSize = calculateInSampleSize(options, 1920, 1920)
    options.inJustDecodeBounds = false
    // 使用 inMutable 以允许后续 recycle
    options.inMutable = true

    return context.contentResolver.openInputStream(uri)?.use { stream ->
        BitmapFactory.decodeStream(stream, null, options)
    }
}

fun calculateInSampleSize(options: BitmapFactory.Options, reqWidth: Int, reqHeight: Int): Int {
    var inSampleSize = 1
    if (options.outHeight > reqHeight || options.outWidth > reqWidth) {
        val halfHeight = options.outHeight / 2
        val halfWidth = options.outWidth / 2
        while (halfHeight / inSampleSize >= reqHeight && halfWidth / inSampleSize >= reqWidth) {
            inSampleSize *= 2
        }
    }
    return inSampleSize
}
```
